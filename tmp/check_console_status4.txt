if [ -f /tmp/fleetbase-console-build.pid ]; then
  pid=$(cat /tmp/fleetbase-console-build.pid)
  echo PID:$pid
  ps -p "$pid" -o pid=,stat=,etime=,pcpu=,pmem=,cmd= || true
else
  echo PID:missing
fi
echo ---
pgrep -af 'pnpm build|ember build|node .*ember|node .*broccoli|node .*babel' || true
echo ---
if [ -f /tmp/fleetbase-console-build.log ]; then
  stat -c 'LOG_SIZE:%s LOG_MTIME:%y' /tmp/fleetbase-console-build.log
  echo ---
  tail -n 100 /tmp/fleetbase-console-build.log
  echo ---
  grep -niE 'error|fail|aborted|exception|postcss|selector|cannot be transformed|killed|out of memory|fatal|unexpected' /tmp/fleetbase-console-build.log | tail -n 50 || true
else
  echo LOG:missing
fi
echo ---
if [ -d /srv/fleetbase/console/dist ]; then
  echo console_dist_ready
else
  echo console_dist_missing
fi
